## 二、渲染优化

### 2.1 Camera设置

```csharp
// 透明窗口专用Camera配置
public class TransparentWindowCamera : MonoBehaviour
{
    [SerializeField] private bool _hdr = false;
    [SerializeField] private bool _postProcessing = false;
    [SerializeField] private Color _backgroundColor = new Color(0, 0, 0, 0);

    private void Awake()
    {
        var camera = GetComponent<Camera>();
        camera.allowHDR = _hdr;
        camera.allowMSAA = false;

        // 清除Alpha
        camera.clearFlags = CameraClearFlags.SolidColor;
        camera.backgroundColor = _backgroundColor;

        // 禁用后处理
        if (!_postProcessing)
        {
            var postLayer = LayerMask.NameToLayer("PostProcessing");
            if (postLayer != -1)
            {
                camera.cullingMask &= ~(1 << postLayer);
            }
        }
    }
}
```

## 三、点击穿透

### 3.1 穿透层配置

```csharp
public class TransparentWindowInput : MonoBehaviour
{
    [SerializeField] private LayerMask _penetrateLayer;
    [SerializeField] private Camera _uiCamera;

    private void Update()
    {
        // 检测点击是否在穿透区域
        if (Input.GetMouseButtonDown(0))
        {
            var worldPos = _uiCamera.ScreenToWorldPoint(Input.mousePosition);
            var hit = Physics2D.Raycast(worldPos, Vector2.zero, 0, _penetrateLayer);

            if (hit.collider != null)
            {
                // 通知UniWindowController穿透点击
                UniWindowController.PenetrateClick();
            }
        }
    }
}
```

## 四、缩放与DPI

### 4.1 DPI适配

```csharp
public class DPIAdapter : MonoBehaviour
{
    private float _scaleFactor;

    private void Start()
    {
        _scaleFactor = Screen.dpi / 96f;
        ApplyScale();
    }

    private void ApplyScale()
    {
        // 根据DPI调整UI缩放
        if (_scaleFactor > 1.5f)
        {
            // 高DPI设备
            UIManager.Instance.SetUIScale(1.5f);
        }
        else if (_scaleFactor > 1.2f)
        {
            // 中等DPI
            UIManager.Instance.SetUIScale(1.2f);
        }
        else
        {
            // 标准DPI
            UIManager.Instance.SetUIScale(1.0f);
        }
    }
}
```

### 4.2 窗口大小变化处理

```csharp
public class WindowResizeHandler : MonoBehaviour
{
    private Vector2Int _lastSize;

    private void Start()
    {
        _lastSize = new Vector2Int(Screen.width, Screen.height);
        Screen.sleepTimeout = SleepTimeout.NeverSleep;
    }

    private void Update()
    {
        if (Screen.width != _lastSize.x || Screen.height != _lastSize.y)
        {
            OnWindowResized();
        }
    }

    private void OnWindowResized()
    {
        _lastSize = new Vector2Int(Screen.width, Screen.height);

        // 通知UI重新布局
        UIManager.Instance.OnResolutionChanged(_lastSize);

        // 重新设置透明背景
        UniWinCore.ResizeWindow(_lastSize.x, _lastSize.y);
    }
}
```

## 五、后台运行

### 5.1 焦点丢失处理

```csharp
public class BackgroundRunningManager : MonoBehaviour
{
    [SerializeField] private bool _pauseOnFocusLost = false;

    private void OnApplicationFocus(bool hasFocus)
    {
        if (!hasFocus && !_pauseOnFocusLost)
        {
            // 后台继续运行
            Time.timeScale = 1f;
            Application.runInBackground = true;
        }
        else if (!hasFocus && _pauseOnFocusLost)
        {
            // 暂停游戏
            Time.timeScale = 0f;
        }
        else
        {
            // 恢复
            Time.timeScale = 1f;
        }
    }

    private void OnApplicationPause(bool pauseStatus)
    {
        if (pauseStatus)
        {
            // 保存进度
            SaveManager.Instance.AutoSave();
        }
    }
}
```

### 5.2 低功耗模式

```csharp
public class PowerSavingMode : MonoBehaviour
{
    [SerializeField] private int _targetFPS = 60;
    [SerializeField] private int _backgroundFPS = 15;

    private bool _isBackground;

    private void Update()
    {
        if (_isBackground)
        {
            // 后台降低帧率省电
            Application.targetFrameRate = _backgroundFPS;
        }
        else
        {
            Application.targetFrameRate = _targetFPS;
        }
    }

    private void OnApplicationFocus(bool hasFocus)
    {
        _isBackground = !hasFocus;
    }
}
```

## 六、输入优化

### 6.1 全局输入钩子

```csharp
// GlobalKeyboardHook.cs 使用Windows API
public class GlobalKeyboardHook
{
    private delegate IntPtr LowLevelKeyboardProc(int nCode, IntPtr wParam, IntPtr lParam);

    [DllImport("user32.dll", CharSet = CharSet.Auto, SetLastError = true)]
    private static extern IntPtr SetWindowsHookEx(int idHook, LowLevelKeyboardProc lpfn, IntPtr hMod, uint dwThreadId);

    [DllImport("user32.dll")]
    [return: MarshalAs(UnmanagedType.Bool)]
    private static extern bool UnhookWindowsHookEx(IntPtr hhk);

    private IntPtr _hookId = IntPtr.Zero;

    public void Start()
    {
        using (var curProcess = Process.GetCurrentProcess())
        using (var curModule = curProcess.MainModule)
        {
            _hookId = SetWindowsHookEx(WH_KEYBOARD_LL, HookCallback, curModule.BaseAddress, 0);
        }
    }

    public void Stop()
    {
        if (_hookId != IntPtr.Zero)
        {
            UnhookWindowsHookEx(_hookId);
            _hookId = IntPtr.Zero;
        }
    }
}
```

### 6.2 输入转换

```csharp
public class InputConverter
{
    private const int KEY_HITS_FOR_1_TP = 10;
    private const float MOUSE_PIXELS_FOR_01_TP = 100f;
    private const float COMBO_WINDOW = 1f;
    private const int COMBO_HITS_FOR_01_BONUS = 10;
    private const float MAX_BONUS = 1.5f;

    private int _keyHitCount;
    private float _mouseMoveTotal;
    private float _lastInputTime;
    private int _comboCount;
    private float _currentBonus = 1f;

    public float ConvertToTravelPoints()
    {
        float tp = 0f;

        // 键盘转TP
        tp += _keyHitCount / KEY_HITS_FOR_1_TP;
        _keyHitCount = 0;

        // 鼠标转TP
        tp += _mouseMoveTotal / MOUSE_PIXELS_FOR_01_TP;
        _mouseMoveTotal = 0f;

        // 连击加成
        if (Time.time - _lastInputTime < COMBO_WINDOW)
        {
            _comboCount++;
            _currentBonus = Mathf.Min(1f + (_comboCount / COMBO_HITS_FOR_01_BONUS) * 0.1f, MAX_BONUS);
        }
        else
        {
            _comboCount = 0;
            _currentBonus = 1f;
        }

        return tp * _currentBonus;
    }

    public void OnKeyPressed()
    {
        _keyHitCount++;
        _lastInputTime = Time.time;
    }

    public void OnMouseMoved(Vector2 delta)
    {
        _mouseMoveTotal += delta.magnitude;
        _lastInputTime = Time.time;
    }
}
```

### 7.2 性能监控

```csharp
public class PerformanceMonitor : MonoBehaviour
{
    private float _fps;
    private float _frameTime;
    private int _drawCalls;
    private long _memory;

    private void LateUpdate()
    {
        _fps = 1f / Time.deltaTime;
        _frameTime = Time.deltaTime * 1000f;
        _drawCalls = UnityEngine.Object.FindObjectsOfType<MeshRenderer>().Length;
        _memory = Profiler.GetTotalAllocatedMemory() / 1024 / 1024;
    }

    public string GetReport()
    {
        return $"FPS: {_fps:F1} | Frame: {_frameTime:F1}ms | DrawCalls: {_drawCalls} | Memory: {_memory}MB";
    }
}
```
