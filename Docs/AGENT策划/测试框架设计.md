## 一、测试架构概述

### 1.1 测试金字塔

```
         ┌─────────────────────┐
         │     E2E 测试 (5%)    │  ← 完整游戏流程
         ├─────────────────────┤
         │   集成测试 (25%)     │  ← 模块间交互
         ├─────────────────────┤
         │     单元测试 (70%)    │  ← 核心逻辑
         └─────────────────────┘
```

### 1.2 测试原则

| 原则 | 说明 |
|------|------|
| AAA模式 | Arrange (准备) → Act (执行) → Assert (断言) |
| 单一职责 | 每个测试只验证一个行为 |
| 独立性 | 测试之间无依赖，可任意顺序执行 |
| 可重复 | 相同输入产生相同输出 |
| 快速执行 | EditMode优先于PlayMode |

        }
        
        [UnityTest]
        public IEnumerator Subscribe_ShouldReceiveEvents()

            // Arrange
            _goldEventBus.Subscribe(OnGoldChanged);
            
            // Act
            _goldEventBus.Publish(new PlayerGoldChangedEvent { NewGold = 100 });
            
            // Assert
            Assert.AreEqual(1, _callbackCount);
            
            yield return null;
        }
        
        [UnityTest]
        public IEnumerator MultipleSubscribers_ShouldReceiveAll()
        {
            // Arrange
            _goldEventBus.Subscribe(OnGoldChanged);
            _goldEventBus.Subscribe(OnGoldChanged);
            
            // Act
            _goldEventBus.Publish(new PlayerGoldChangedEvent { NewGold = 100 });
            
            // Assert
            Assert.AreEqual(2, _callbackCount);
            
            yield return null;
        }
        
        [UnityTest]
        public IEnumerator Unsubscribe_ShouldNotReceiveEvents()
        {
            // Arrange
            _goldEventBus.Subscribe(OnGoldChanged);
            _goldEventBus.Unsubscribe(OnGoldChanged);
            
            // Act
            _goldEventBus.Publish(new PlayerGoldChangedEvent { NewGold = 100 });
            
            // Assert
            Assert.AreEqual(0, _callbackCount);
            
            yield return null;
        }
        
        private void OnGoldChanged(PlayerGoldChangedEvent evt)
        {
            _callbackCount++;
        }
    }

            _context.isCritical = false;
            
            // Act
            var damage = _combatSystem.CalculateDamage(_context);
            
            // Assert
            Assert.Less(damage, 100, "防御应该减少伤害");

        public IEnumerator CalculateDamage_ShouldApplyCritMultiplier()
        {

            _context.defender = CreateTestCombatant(defense: 0);
            _context.isCritical = true;
            _context.critMultiplier = 2.0f;
            
            // Act
            var normalDamage = _combatSystem.CalculateDamage(_context, isCritical: false);
            var critDamage = _combatSystem.CalculateDamage(_context, isCritical: true);
            
            // Assert
            Assert.AreEqual(normalDamage * 2.0f, critDamage);
            
            yield return null;
        }
        
        [UnityTest]
        public IEnumerator ExecuteAttack_ShouldTriggerEvent()
        {
            // Arrange
            var attacker = CreateTestCombatant(attack: 50);
            var defender = CreateTestCombatant(defense: 10, hp: 100);

            _combatSystem.OnDamageDealt += (sender, args) => eventFired = true;
            
            // Act
            _combatSystem.ExecuteAttack(attacker, defender, out var damage);
            
            // Assert
            Assert.IsTrue(eventFired, "伤害事件应该触发");
            Assert.Greater(damage, 0, "伤害应该大于0");
            
            yield return null;
        }
