# 成就系统设计文档

## 1. 概述

### 1.1 设计目标
为Game1挂机放置RPG游戏添加完整的成就/任务系统，提供玩家长期目标感和留存驱动。

### 1.2 设计原则
- 遵循项目现有Design+Manager模块设计模式
- 采用EventBus事件驱动架构
- 支持存档持久化
- 复用现有DataBinding UI系统

### 1.3 参考借鉴
- **Melvor Idle成就树**：分类体系+前置依赖树形结构
- **Realm Grinder Trophy系统**：里程碑式成就+累积进度追踪
- **项目现有模块**：InventoryDesign/TeamDesign等单例设计

### 4.2 触发机制流程

`
1. 游戏事件发生 (如 GoldChanged)
         ↓
2. EventBus.Publish(new GoldChangedEvent(...))
         ↓
3. AchievementDesign.OnGoldChanged(event)
         ↓
4. 遍历所有未解锁成就
         ↓
5. 检查是否有成就包含 GoldEarned 条件
         ↓
6. 更新 conditionProgress
         ↓
7. 如果所有条件满足 → 解锁成就 → 发布 AchievementUnlockedEvent
         ↓
8. UI响应 AchievementUnlockedEvent → 显示解锁动画
`

### 4.3 AchievementDesign事件处理

`csharp
public class AchievementDesign
{
    #region EventBus Subscription
    public void Initialize()
    {
        EventBus.Subscribe<GoldChangedEvent>(OnGoldChanged);
        EventBus.Subscribe<LevelUpEvent>(OnLevelUp);
        EventBus.Subscribe<TravelCompletedEvent>(OnTravelCompleted);
        EventBus.Subscribe<CombatResultEvent>(OnCombatResult);
        EventBus.Subscribe<ItemAcquiredEvent>(OnItemAcquired);
        EventBus.Subscribe<TeamMemberAddedEvent>(OnTeamMemberAdded);
        EventBus.Subscribe<EventTreeCompletedEvent>(OnEventTreeCompleted);
    }
    
    public void Dispose()
    {
        EventBus.Unsubscribe<GoldChangedEvent>(OnGoldChanged);
        // ... 其他事件
    }
    #endregion
    
    #region Event Handlers
    private void OnGoldChanged(GoldChangedEvent e)
    {
        UpdateConditionProgress(AchievementConditionType.GoldEarned, e.delta);
    }
    
    private void OnTravelCompleted(TravelCompletedEvent e)
    {
        UpdateConditionProgress(AchievementConditionType.DistanceTraveled, e.distance);
        UpdateConditionProgress(AchievementConditionType.TravelCount, 1);
    }
    
    private void OnCombatResult(CombatResultEvent e)
    {
        if (e.victory)
        {
            UpdateConditionProgress(AchievementConditionType.CombatWinCount, 1);
            UpdateConditionProgress(AchievementConditionType.EnemiesDefeated, e.enemiesDefeated);
        }
        if (e.noDamageTaken)
        {
            UpdateConditionProgress(AchievementConditionType.CombatNoDamageCount, 1);
        }
    }
    
    // ... 其他处理器
    #endregion
    
    #region Progress Update
    private void UpdateConditionProgress(AchievementConditionType type, float value)
    {
        foreach (var instance in _unlockedAchievements.Values)
            continue;  // 已解锁的不再检查
        
        foreach (var instance in _achievements.Values)
        {
            if (instance.isUnlocked) continue;
            
            var template = GetTemplate(instance.templateId);
            for (int i = 0; i < template.conditions.Count; i++)
            {
                if (template.conditions[i].type == type)
                {
                    instance.conditionProgress[i] += value;
                    CheckAndUnlock(instance);
                }
            }
        }
    }
    
    private void CheckAndUnlock(AchievementInstance instance)
    {
        var template = GetTemplate(instance.templateId);
        
        // 检查所有条件是否满足
        bool allConditionsMet = true;
        for (int i = 0; i < template.conditions.Count; i++)
        {
            if (instance.conditionProgress[i] < template.conditions[i].targetValue)
            {
                allConditionsMet = false;
                break;
            }
        }
        
        if (allConditionsMet)
        {
            UnlockAchievement(instance);
        }
    }
    
    private void UnlockAchievement(AchievementInstance instance)
    {
        instance.isUnlocked = true;
        instance.unlockTime = DateTimeOffset.UtcNow.ToUnixTimeSeconds();
        
        // 发放奖励
        var template = GetTemplate(instance.templateId);
        foreach (var reward in template.rewards)
        {
            GrantReward(reward);
        }
        
        // 发布解锁事件
        EventBus.Publish(new AchievementUnlockedEvent
        {
            achievementId = instance.templateId,
            unlockTime = instance.unlockTime
        });
    }
    #endregion
}
`

public class UIAchievementPanel : BaseUIPanel
{
    #region UI Components
    [SerializeField] private Transform categoryTabContainer;
    [SerializeField] private Transform achievementListContainer;
    [SerializeField] private UIAchievementItem itemPrefab;
    [SerializeField] private EventDrivenProgressBar totalProgressBar;
    #endregion
    
    #region DataBinding
    private AchievementDataSource _dataSource;
    
    protected override void OnInitialize()
    {
        base.OnInitialize();
        _dataSource = new AchievementDataSource(AchievementManager.GetAllAchievements());
        
        // 绑定分类标签
        categoryTabContainer.SetData(_dataSource.Categories, OnCategorySelected);
        
        // 绑定成就列表
        achievementListContainer.SetData(_dataSource.CurrentCategoryAchievements, itemPrefab);
        
        // 绑定总体进度
        totalProgressBar.SetSource(_dataSource.TotalProgressSource);
        
        // 监听解锁事件
        EventBus.Subscribe<AchievementUnlockedEvent>(OnAchievementUnlocked);
    }
    #endregion
    
    private void OnCategorySelected(AchievementCategory category)
    {
        _dataSource.SetCategory(category);
        achievementListContainer.Refresh();
    }
    
    private void OnAchievementUnlocked(AchievementUnlockedEvent e)
    {
        // 显示解锁动画/特效
        PlayUnlockEffect(e.achievementId);
    }
}

public class UIAchievementItem : BaseDataSourceItem<AchievementInstance>
{
    [SerializeField] private Image icon;
    [SerializeField] private UIText nameText;
    [SerializeField] private UIText descText;
    [SerializeField] private EventDrivenProgressBar progressBar;
    [SerializeField] private GameObject lockOverlay;
    [SerializeField] private GameObject unlockedBadge;
    
    public override void SetData(AchievementInstance data)
    {
        base.SetData(data);
        
        var template = AchievementManager.GetTemplate(data.templateId);
        
        nameText.SetKey(template.nameId);
        descText.SetKey(template.descriptionId);
        icon.sprite = ResourceManager.Load<Sprite>(template.iconId);
        
        if (data.isUnlocked)
        {
            lockOverlay.SetActive(false);
            unlockedBadge.SetActive(true);
            progressBar.gameObject.SetActive(false);
        }
        else
        {
            lockOverlay.SetActive(true);
            unlockedBadge.SetActive(false);
            progressBar.gameObject.SetActive(true);
            progressBar.SetProgress(data.GetProgressPercent());
        }
    }
}

### 8.3 TaskDesign刷新逻辑

`csharp
public class TaskDesign
{
    private const int DAILY_TASK_COUNT = 5;
    private const int WEEKLY_TASK_COUNT = 3;
    
    public void CheckDailyRefresh()
    {
        var now = DateTimeOffset.Now.ToUnixTimeSeconds();
        var lastRefresh = GetLastDailyRefreshTime();
        
        // 检查是否跨天（北京时间00:00）
        if (IsNewDay(lastRefresh, now))
        {
            RefreshDailyTasks();
            SetLastDailyRefreshTime(now);
        }
    }
    
    public void CheckWeeklyRefresh()
    {
        var now = DateTimeOffset.Now.ToUnixTimeSeconds();
        var lastRefresh = GetLastWeeklyRefreshTime();
        
        // 检查是否跨周（北京时间周一00:00）
        if (IsNewWeek(lastRefresh, now))
        {
            RefreshWeeklyTasks();
            SetLastWeeklyRefreshTime(now);
        }
    }
    
    private void RefreshDailyTasks()
    {
        // 从模板库随机抽取DAILY_TASK_COUNT个任务
        var templates = GetAvailableTemplates(AchievementCategory.TaskDaily);
        var selected = RandomSelect(templates, DAILY_TASK_COUNT);
        
        _currentDailyTasks.Clear();
        foreach (var t in selected)
        {
            _currentDailyTasks.Add(CreateInstance(t));
        }
        
        EventBus.Publish(new DailyTasksRefreshedEvent());
    }
}
`
