## 三、内存管理

### 3.1 对象池策略

```csharp
public class ObjectPool<T> where T : Component
{
    private Queue<T> _available = new Queue<T>();
    private HashSet<T> _inUse = new HashSet<T>();
    private T _prefab;
    private Transform _parent;

    public ObjectPool(T prefab, int initialSize, Transform parent)
    {
        _prefab = prefab;
        _parent = parent;

        for (int i = 0; i < initialSize; i++)
        {
            var obj = UnityEngine.Object.Instantiate(prefab, parent);
            obj.gameObject.SetActive(false);
            _available.Enqueue(obj);
        }
    }

    public T Get(Vector3 position, Quaternion rotation)
    {
        T obj;

        if (_available.Count > 0)
        {
            obj = _available.Dequeue();
        }
        else
        {
            obj = UnityEngine.Object.Instantiate(_prefab, _parent);
        }

        obj.transform.SetPositionAndRotation(position, rotation);
        obj.gameObject.SetActive(true);
        _inUse.Add(obj);

        return obj;
    }

    public void Return(T obj)
    {
        if (!_inUse.Contains(obj)) return;

        obj.gameObject.SetActive(false);
        _inUse.Remove(obj);
        _available.Enqueue(obj);
    }
}
```

### 3.2 资源生命周期

```csharp
public class ResourceLifecycleManager
{
    private Dictionary<string, ResourceHandle> _loadedResources = new Dictionary<string, ResourceHandle>();
    private ReferenceCountedPool _referencePool = new ReferenceCountedPool();

    public T Load<T>(string path) where T : UnityEngine.Object
    {
        if (_loadedResources.TryGetValue(path, out var handle))
        {
            handle.AddRef();
            return handle.Asset as T;
        }

        var asset = Resources.Load<T>(path);
        handle = new ResourceHandle(asset);
        _loadedResources[path] = handle;

        return asset;
    }

    public void Unload(string path)
    {
        if (!_loadedResources.TryGetValue(path, out var handle)) return;

        handle.Release();
        if (handle.CanUnload)
        {
            Resources.UnloadAsset(handle.Asset);
            _loadedResources.Remove(path);
        }
    }
}
```

### 3.3 内存泄漏检测

```csharp
public class MemoryLeakDetector
{
    private List<ILeakCheckable> _trackedObjects = new List<ILeakCheckable>();
    private Action<string> _onLeakDetected;

    public void Track(ILeakCheckable obj, string description)
    {
        _trackedObjects.Add(obj);
        obj.OnDisposed += () => OnObjectDisposed(obj);
    }

    public void CheckForLeaks()
    {
        var leaked = _trackedObjects.Where(o => !o.IsValid).ToList();
        foreach (var leak in leaked)
        {
            _onLeakDetected?.Invoke($"Leak: {leak.Description}");
        }
    }
}

public interface ILeakCheckable
{
    bool IsValid { get; }
    string Description { get; }
    event Action OnDisposed;
}
```

### 4.2 异步加载策略

```csharp
public class AsyncLoadingManager
{
    private List<AsyncOperation> _pendingOperations = new List<AsyncOperation>();
    private float _loadingProgress;

    public void LoadSceneAdditive(string sceneAddress)
    {
        var handle = Addressables.LoadSceneAsync(sceneAddress, UnityEngine.SceneManagement.LoadSceneMode.Additive);
        _pendingOperations.Add(handle);

        handle.Completed += op =>
        {
            _pendingOperations.Remove(handle);
            OnSceneLoaded(sceneAddress, op);
        };
    }

    public void LoadUI<T>(string uiAddress, Action<T> onLoaded) where T : Component
    {
        var handle = Addressables.LoadAssetAsync<GameObject>(uiAddress);
        _pendingOperations.Add(handle);

        handle.Completed += op =>
        {
            var prefab = op.Result;
            var instance = UnityEngine.Object.Instantiate(prefab);
            var component = instance.GetComponent<T>();
            onLoaded?.Invoke(component);
        };
    }

    public float GetOverallProgress()
    {
        if (_pendingOperations.Count == 0) return 1f;

        float total = 0f;
        foreach (var op in _pendingOperations)
        {
            total += op.progress;
        }
        return total / _pendingOperations.Count;
    }
}
```

```csharp
public class DirtyFlagSystem
{
    private Dictionary<int, bool> _dirtyFlags = new Dictionary<int, bool>();

    public void SetDirty(int id)
    {
        _dirtyFlags[id] = true;
    }

    public void ProcessDirty<T>(IEnumerable<T> entities, Action<T> updateFunc) where T : IDirtyable
    {
        foreach (var entity in entities)

            if (entity.IsDirty)
```

### 5.3 批处理更新

```csharp
public class BatchProcessor<T>
{
    private List<T> _pending = new List<T>();
    private Action<T> _processAction;
    private int _batchSize = 100;

    public BatchProcessor(Action<T> processAction)
    {
        _processAction = processAction;
    }

    public void Add(T item)
    {
        _pending.Add(item);
    }

    public void ProcessAll()
    {
        int processed = 0;
        while (processed < _pending.Count)
        {
            int count = Mathf.Min(_batchSize, _pending.Count - processed);
            for (int i = 0; i < count; i++)
            {
                _processAction(_pending[processed + i]);
            }
            processed += count;

            // 如果有剩余，让出主线程
            if (processed < _pending.Count)
            {
                yield return null;
            }
        }

        _pending.Clear();
    }
}
```

```csharp
    {
        _pendingDeltas.Add(new SaveDelta
        {
            Key = key,
            OldValue = oldValue,
            NewValue = newValue,
            Timestamp = DateTime.Now
        });
    }

    public GameSaveData GenerateIncrementalSave()
    {
        var save = new GameSaveData();

        // 应用基准
        ApplyBase(save);

        // 应用增量
        foreach (var delta in _pendingDeltas)
        {
            ApplyDelta(save, delta);
        }

        return save;
    }

    public void MergeIncrementalSave(GameSaveData incremental)
    {
        // 合并增量到基准

    }
}
```

### 6.2 压缩存档

```csharp
public class CompressedSaveSystem
{
    public string CompressAndEncode(GameSaveData data)
    {
        var json = JsonUtility.ToJson(data);
        var bytes = System.Text.Encoding.UTF8.GetBytes(json);

        using (var output = new MemoryStream())
        {
            using (var gzip = new System.IO.Compression.GZipStream(output, System.IO.Compression.CompressionMode.Compress))
            {
                gzip.Write(bytes, 0, bytes.Length);
            }
            return Convert.ToBase64String(output.ToArray());
        }
    }

    public GameSaveData DecodeAndDecompress(string encoded)
    {
        var bytes = Convert.FromBase64String(encoded);

        using (var input = new MemoryStream(bytes))
        using (var gzip = new System.IO.Compression.GZipStream(input, System.IO.Compression.CompressionMode.Decompress))
        using (var output = new MemoryStream())
        {
            gzip.CopyTo(output);
            var json = System.Text.Encoding.UTF8.GetString(output.ToArray());
            return JsonUtility.FromJson<GameSaveData>(json);
        }
    }
}
```
