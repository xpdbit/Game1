```csharp
// SaveManager.Save() 实现
public void Save()
{
    try
    {
        // 确保目录存在
        string path = GetSavePath();
        string directory = Path.GetDirectoryName(path);
        if (!Directory.Exists(directory))
        {
            Directory.CreateDirectory(directory);
        }

        // 更新时间戳
        _currentSave.timestamp = DateTime.Now.Ticks;

        // 序列化并写入
        string json = JsonUtility.ToJson(_currentSave, true);
        File.WriteAllText(path, json);

        _isDirty = false;
        Debug.Log($"[SaveManager] Game saved to: {path}");
    }
    catch (Exception ex)
    {
        Debug.LogError($"[SaveManager] Save failed: {ex.Message}");
    }
}

// SaveManager.Load() 实现
public void Load()
{
    try
    {
        string path = GetSavePath();
        if (File.Exists(path))
        {
            string json = File.ReadAllText(path);
            _currentSave = JsonUtility.FromJson<GameSaveData>(json);

            if (_currentSave == null)
            {
                Debug.LogWarning("[SaveManager] Save file corrupted, creating new save");
                CreateNewSave();
            }
            else
            {
                Debug.Log($"[SaveManager] Game loaded from: {path}");
                Debug.Log($"[SaveManager] Last played: {new DateTime(_currentSave.timestamp)}");
            }
        }
        else
        {
            Debug.Log("[SaveManager] No save file found, creating new game");
            CreateNewSave();
        }
    }
    catch (Exception ex)
    {
        Debug.LogError($"[SaveManager] Load failed: {ex.Message}");
        CreateNewSave();
    }
}

// SaveManager.CreateNewSave() 完整实现
public void CreateNewSave()
{
    _currentSave = new GameSaveData
    {
        player = new PlayerSaveData
        {

        },
        world = new WorldSaveData
        {

            timestamp = DateTime.Now.Ticks
        },
        playTime = 0,
        timestamp = DateTime.Now.Ticks
    };
    _isDirty = true; // 新游戏也需要标记以便自动存档
```

**任务2.2：修复SaveManager自动存档逻辑**

```csharp
// 问题：首次运行从未dirty，从不自动存档
// 当前代码SaveManager.cs:117-125
```

**修改为：**

```csharp
public void Tick(float deltaTime)
{
    // 始终计时，不只是dirty时
    _autoSaveTimer += deltaTime;

    // 无论是否dirty，都定期存档（每5分钟）
    // 这样可以记录玩家在线时间
    if (_autoSaveTimer >= _autoSaveInterval)
    {
        // 如果有变化才实际写入
        if (_isDirty)
        {
            Save();
        }
        else
        {
            // 即使没变化，也更新时间戳
            _currentSave.timestamp = DateTime.Now.Ticks;
        }
        _autoSaveTimer = 0f;
    }
}

// 同时在关键操作后调用MarkDirty()
public void MarkDirty()
{
    _isDirty = true;
}
```

```csharp
public float CalculateOfflineReward(float offlineSeconds)
{
    // 1. 上限24小时
    float cappedSeconds = Mathf.Min(offlineSeconds, 24f * 3600f);

    // 2. 护肝机制：超过6小时收益减半
    float rateMultiplier = cappedSeconds > 6f * 3600f ? 0.5f : 1.0f;

    // 3. 计算收益
    float reward = cappedSeconds * baseRewardPerSecond * offlineRewardRate * GetCurrentRewardRate() * rateMultiplier;

    return reward;
}
```

#### Day 9：成长曲线优化

**任务9.1：实现分段式升级经验曲线**

```csharp
// PlayerActor.cs 或单独的 LevelSystem.cs
public static class LevelSystem
{
    /// <summary>
    /// 计算升级所需经验
    /// </summary>
    /// <param name="currentLevel">当前等级</param>
    /// <returns>升到下一级所需经验</returns>
    public static int GetExpForLevel(int currentLevel)
    {
        if (currentLevel <= 20)
        {
            // 前20级：每级 +10%
            return Mathf.FloorToInt(10 * Mathf.Pow(currentLevel, 1.0f));
        }
        else if (currentLevel <= 40)
        {
            // 21-40级：每级 +8%
            return Mathf.FloorToInt(10 * Mathf.Pow(currentLevel, 1.1f));
        }
        else
        {
            // 41级后：每级 +5%
            return Mathf.FloorToInt(10 * Mathf.Pow(currentLevel, 1.2f));
        }
    }

    /// <summary>
    /// 计算暴击率
    /// </summary>
    public static float CalculateCritRate(int wisdom)
    {
        return Mathf.Min(0.5f, wisdom / 100f);
    }

    /// <summary>
    /// 计算伤害减免
    /// </summary>
    public static float CalculateDamageReduction(int defense)
    {
        return defense / (defense + 100f);
    }
}
```

### A. SaveManager.Save() 完整修改

```csharp
// ===== 修改前 =====
public void Save()
{
    // TODO: 实现存档逻辑
    _isDirty = false;
}

// ===== 修改后 =====
public void Save()
{
    try
    {
        string path = GetSavePath();
        string directory = Path.GetDirectoryName(path);

        if (!Directory.Exists(directory))
            Directory.CreateDirectory(directory);

        _currentSave.timestamp = DateTime.Now.Ticks;
        string json = JsonUtility.ToJson(_currentSave, true);
        File.WriteAllText(path, json);

        _isDirty = false;
        Debug.Log($"[SaveManager] Game saved to: {path}");
    }
    catch (Exception ex)
    {
        Debug.LogError($"[SaveManager] Save failed: {ex.Message}");
    }
}
```

### B. GameMain.InitializeManagers() 完整修改

```csharp
// ===== 修改前 =====
private void InitializeManagers()
{
    if (travelManager == null)
        travelManager = TravelManager.instance;
    if (progressManager == null)
        progressManager = ProgressManager.instance;
    if (eventQueue == null)
        eventQueue = new EventQueue();
    if (eventChainManager == null)
        eventChainManager = EventChainManager.instance;
    if (npcManager == null)
        npcManager = NPCManager.instance;
    if (combatSystem == null)
        combatSystem = CombatSystem.instance;
    travelManager?.SetEventQueue(eventQueue);
}

// ===== 修改后 =====
private void InitializeManagers()
{
    // GameLoopManager必须首先创建
    if (gameLoopManager == null)
        gameLoopManager = gameObject.AddComponent<GameLoopManager>();

    // 其他管理器
    if (travelManager == null)
        travelManager = TravelManager.instance;
    if (progressManager == null)
        progressManager = ProgressManager.instance;
    if (eventQueue == null)
        eventQueue = new EventQueue();
    if (eventChainManager == null)
        eventChainManager = EventChainManager.instance;
    if (npcManager == null)
        npcManager = NPCManager.instance;
    if (combatSystem == null)
        combatSystem = CombatSystem.instance;

    // GameLoopManager初始化完成后获取PlayerActor
    if (playerActor == null && GameLoopManager.instance != null)
        playerActor = GameLoopManager.instance.player;

    travelManager?.SetEventQueue(eventQueue);
}
```
